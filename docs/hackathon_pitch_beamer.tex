% 3) Problème
\begin{frame}{Le problème concret}
\begin{block}{Cas typique : application de gestion de tâches}
\begin{itemize}
  \item Le métier donne un besoin en texte/Markdown.
  \item L'équipe doit livrer \textbf{plan + architecture + app + tests + CI}.
  \item Chaque livrable est produit séparément, souvent sans cohérence globale.
\end{itemize}
\end{block}
\vspace{0.2cm}
\begin{itemize}
  \item Multiplication des allers-retours.
  \item Risque de dérive entre intention métier et code.
  \item Difficile à auditer pour un contexte entreprise.
\end{itemize}
\end{frame}

% 4) Enjeux
\section{Enjeux}
\begin{frame}{Pourquoi c'est critique ?}
\begin{columns}[T]
\column{0.5\textwidth}
\textbf{Enjeux business}
\begin{itemize}
  \item Time-to-prototype
  \item Coût de coordination
  \item Qualité perçue en démo
\end{itemize}
\column{0.5\textwidth}
\textbf{Enjeux techniques}
\begin{itemize}
  \item Cohérence inter-artefacts
  \item Gouvernance des décisions
  \item Reproductibilité des runs
\end{itemize}
\end{columns}
\vspace{0.4cm}
\begin{alertblock}{Contrainte hackathon IBM}
Démontrer une solution multi-agents \textbf{défendable en entreprise} : traçable, lisible, fiable.
\end{alertblock}
\end{frame}

% 5) Solution proposée
\section{Solution}
\begin{frame}{Notre solution}
\begin{block}{AI Delivery Multi-Agent}
Un \textbf{générateur Python multi-agents} qui transforme une exigence en artefacts clés d'une chaîne logicielle.
\end{block}
\vspace{0.2cm}
\begin{itemize}
  \item Entrées : texte, Markdown, JSON.
  \item Sorties : analyse, plan, C4, code app, tests, CI/CD, traces Plan/Act/Reason.
  \item Exécution via CLI \textbf{et} API.
\end{itemize}
\vspace{0.2cm}
\begin{center}
\textbf{Objectif :} industrialiser la phase \og de l'idée au prototype crédible \fg.
\end{center}
\end{frame}

% 8) Fonctionnement global
\section{Fonctionnement}
\begin{frame}{Fonctionnement global (pipeline)}
\begin{enumerate}
  \item Ingestion et normalisation des exigences.
  \item Plan projet et design architecture.
  \item Génération de l'application (backend/frontend/templates).
  \item Génération C4 + CI/CD + validations.
  \item Reporting final + traces de raisonnement.
\end{enumerate}
\vspace{0.2cm}
\begin{block}{Contrôle humain}
Mode interactif : \textbf{pause, intervention, reprise} entre étapes critiques.
\end{block}
\end{frame}

% 9) Démonstration concrète
\section{Démonstration}
\begin{frame}{Démonstration (ce que voit le jury)}
\begin{columns}[T]
\column{0.50\textwidth}
\textbf{Walkthrough live en 90 secondes}
\vspace{0.15cm}
\begin{enumerate}
  \item \textbf{On lance} le run\\
  \scriptsize{\texttt{ai\_delivery.cli run --input inputs/sample\_spec.md}}
  \item \textbf{On suit} les étapes agents\\
  \scriptsize{spec\_analyst $\rightarrow$ architect $\rightarrow$ developer...}
  \item \textbf{On ouvre} les artefacts générés\\
  \scriptsize{\texttt{workspace/generated\_app} + \texttt{outputs/*}}
\end{enumerate}
\vspace{0.15cm}
\textbf{Interaction possible :}
\begin{itemize}

% 10) Résultats
\section{Résultats}
\begin{frame}{Résultats obtenus}
\begin{itemize}
  \item Orchestrateur multi-agents fonctionnel (CLI + API).
  \item Génération de l'app de démonstration dans \texttt{workspace/generated\_app/}.
  \item Production de documents C4 (context, container, component).
  \item Export des traces de raisonnement et plans d'exécution.
  \item Suite de tests unitaires + intégration pour sécuriser la démo.
\end{itemize}
\vspace{0.2cm}
\begin{block}{Impact immédiat}
Une équipe passe d'une spécification brute à une base de projet cohérente en un run.
\end{block}
\end{frame}

% 12) Crédibilité/faisabilité
\section{Crédibilité}
\begin{frame}{Crédibilité et faisabilité}
\begin{itemize}
  \item Choix techniques pragmatiques et éprouvés.
  \item Découplage provider LLM: mock local \& extension Snowflake.
  \item Gouvernance par contrats de données typés.
  \item Exécution déterministe, simple à expliquer au jury.
\end{itemize}
\vspace{0.25cm}
\begin{block}{Pourquoi ça marche ?}
\begin{itemize}
  \item Chaque agent a un périmètre clair.
  \item Les artefacts sont écrits à des emplacements standardisés.
  \item Les tests couvrent les unités et les scénarios d'intégration.
\end{itemize}
\end{block}
\end{frame}
